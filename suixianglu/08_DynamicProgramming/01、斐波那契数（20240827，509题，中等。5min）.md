```C++ {.line-numbers}
class Solution {
public:
    int fib(int n) {
        if(n == 0) return 0;  // 注意这种情况会使下面dp[1] = 1语句出错
        vector<int> dp(n+1, 0);  // 从0到n有n+1个数！！！
        dp[0] = 0;
        dp[1] = 1;
        for(int i = 2; i <= n; i++){
            dp[i] = dp[i-1] + dp[i-2];
        }
        return dp[n];
    }
};
```

```C++ {.line-numbers}
    int fib(int N) {

        dp[0] = 0;
        dp[1] = 1;
        for (int i = 2; i <= N; i++) {
            dp[i] = dp[i - 1] + dp[i - 2];
        }
        return dp[N];
    }
};
```

```C++ {.line-numbers}
class Solution {
public:
    int fib(int N) {
        if (N <= 1) return N;
        int dp[2];
        dp[0] = 0;
        dp[1] = 1;
        for (int i = 2; i <= N; i++) {
            int sum = dp[0] + dp[1];
            dp[0] = dp[1];
            dp[1] = sum;
        }
        return dp[1];
    }
};
```
</div>

时间复杂度：O(n)  
空间复杂度：O(1)
