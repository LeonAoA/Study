```C++
class Solution {
public:
    int rob(vector<int>& nums) {
        vector<int> dp(nums.size()+1, 0);
        dp[0] = 0;

        if(nums.size() == 1) return nums[0];
        if(nums.size() == 2) return max(nums[0], nums[1]);

        dp[0] = 0;
        dp[1] = nums[0];
        dp[2] = max(nums[0], nums[1]);

        for(int j = 3; j <= nums.size(); j++){
            dp[j] = max(dp[j-2] + nums[j-1], dp[j-1]);
        }
        return dp[nums.size()];
    }
};
```
