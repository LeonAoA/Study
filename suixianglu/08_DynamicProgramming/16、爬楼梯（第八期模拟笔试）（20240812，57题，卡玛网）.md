```C++
#include <bits/stdc++.h>

using namespace std;

/*
思路：
将跳1~m阶楼梯看成1~m个物体，
    则问题转化为：用这m个问题填满容量为n的背包有多少种方法
    
dp[j]表示n阶楼梯有dp[j]种走法
    dp[j] = dp[j] + dp[j-k]   k为当前物品价值
    dp[0] = 1
    
楼梯可以重复使用，所以j从小到大遍历
并且由于{1,2}、{2,1}是两种不同的方案，所以是求排列数
    需要j在外层循环
*/
int main(){
    int n,m;
    while(~scanf("%d %d", &n, &m)){
        vector<int> dp(n+1, 0);
        
        dp[0] = 1;
        for(int j = 0; j <= n; j++){
            for(int i = 1; i <= m; i++){
                if(j >= i){
                    dp[j] = dp[j] + dp[j-i];
                }
            }
        }
        printf("%d", dp[n]);
    }
    // printf("%d", dp[n]);
    return 0;
}
```

```C++
#include <iostream>
#include <vector>
using namespace std;
int main() {
    int n, m;
    while (cin >> n >> m) {
        vector<int> dp(n + 1, 0);
        dp[0] = 1;
        for (int i = 1; i <= n; i++) { // 遍历背包
            for (int j = 1; j <= m; j++) { // 遍历物品
                if (i - j >= 0) dp[i] += dp[i - j];
            }
        }
        cout << dp[n] << endl;
    }
}
```
</div>

时间复杂度: O(n * m)  
空间复杂度: O(n)
